MultivariateSAR<-function(DVname, DataFrame, Neighbour, IVlist, Outputname){
	#Spatial Regression Decision tree on dataframe for a dependent variable and multiple independent variables
	#DVname: Name of dependent variable column
	#DataFrame: R data frame/table
	#Neighbour: spatial weights matrix file from Geoda. from spdep read.gal or read.gwt2nb, left as type nb
	#IVlist should resemble "MedAge + BachelorAb + etc."
	#Outputname = "file.txt"
	
	formulaReg<-formula(paste(DVname, "~", IVlist)
	
	#Run Spatial Regression Decision Function
	
	cat(("/nRunning Spatial Regression Decision with ", formulaReg)
	SpatialRegressionDecision(formulaReg, DataFrame, Neighbour, Outputname)
	
}
