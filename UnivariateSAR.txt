UnivariateSAR<-function(DVname, DataFrame, Neighbour, startIVcolnum, endIVcolnum){
	#Spatial Regression Decision tree on dataframe for a dependent variable and a single independent variable
	#DVname: Name of dependent variable column
	#DataFrame: R data frame/table
	#Neighbour: spatial weights matrix file from Geoda. from spdep read.gal or read.gwt2nb, left as type nb
	#startIVcolnum: starting column number for the independent variables to be tested for univariate regression
	#endIVcolnum: ending column number for the independent variables to be tested for univariate regression
		
	for(i in startIVcolnum:endIVcolnum)
	{
		IV<-colnames(DataFrame)[i]
		formulaReg<-formula(paste(DVname, "~",IV))
		
		formulaReg

		OutputName<-paste(DVname,"_x_",IV,".txt", sep="")

		#Run Spatial Regression Decision Function
		
		cat("\nRunning Spatial Regression Decision with: ")
		print(formulaReg)
		cat("\n")
		
		cat("Running Spatial Regression Decision with: ",DVname, "~",IV,file=OutputName, append=FALSE)
		cat("\n\n\n", file=OutputName, append=TRUE)
		
		SpatialRegressionDecision(formulaReg, DataFrame, Neighbour, OutputName)
		
		cat("\n\n\n\n\n\n")
	}
	
}
